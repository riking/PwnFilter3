# Domain Advertising Filter
match ((http)*(\w|\W|\d|_)*(www)*(\w|\W|\d|_)*[a-zA-Z0-9\.\-\*_\^\+\~\`\=\,\&*]{3,}(\W|\d|_|dot|\(dot\))+(com\b|org\b|net\b|edu\b|co\b|uk\b|de\b|cc\b|biz\b|mobi\b|xxx\b|tv\b))
# Please put the name of your server below.
# Website Allow List. Afer yahoo.com, put a | followed by your website name (please
# 	include any punctuation or com, edu, etc after it or it WILL cause problems)
#---------------------------------------------------------------------------------
ignore string youtube.com|google.com|biblegateway.com|winfamilyserver.co.uk|wraithaven.com|minecraft.net|minecraftwiki.com|yahoo.com
#---------------------------------------------------------------------------------
then replace ~^~^~Blocked For Domain Advertising~^~^~
then console say Blocked &player's From Advertising: (Domain)
#then kill Do NOT Advertise!
then log
#
#
#
# IP Address Advertising Filter

#---------------------------------------------------------------------------------
# Please put your IP address of the server in the space below
# After the 1.2.3.4, put a | followed by your server's IP address
#---------------------------------------------------------------------------------
ignore string 192.168.1.1|1.2.3.4
#---------------------------------------------------------------------------------
then replace ~^~^~Blocked For IP Advertising~^~^~
then console say Blocked &player's From Advertising: (IP)
#then kill Do NOT Advertise!
